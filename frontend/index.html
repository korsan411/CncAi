إصلاح وظيفة updateTopView()

تم إصلاح المشكلة في وظيفة updateTopView() لاستخدام بيانات المحاكاة الفعلية بدلاً من المسار الثابت.

// الكود المُصحح: function updateTopView() { const topView = document.getElementById('topView'); if (!topView || !simulation) return; // مسح المحتوى السابق topView.innerHTML = ''; // إنشاء عنصر canvas جديد const canvas = document.createElement('canvas'); canvas.width = 600; canvas.height = 400; topView.appendChild(canvas); const ctx = canvas.getContext('2d'); ctx.fillStyle = '#f0f0f0'; ctx.fillRect(0, 0, canvas.width, canvas.height); if (simulation.points && simulation.points.length > 0) { // رسم مسار الأداة بناءً على بيانات المحاكاة ctx.beginPath(); ctx.strokeStyle = '#4a90e2'; ctx.lineWidth = 2; const firstPoint = simulation.points[0]; ctx.moveTo(firstPoint.x * 5 + 300, firstPoint.y * 5 + 200); for (let i = 1; i < simulation.points.length; i++) { const point = simulation.points[i]; ctx.lineTo(point.x * 5 + 300, point.y * 5 + 200); } ctx.stroke(); // رسم نقطة البداية والنهاية ctx.fillStyle = 'green'; ctx.beginPath(); ctx.arc(firstPoint.x * 5 + 300, firstPoint.y * 5 + 200, 5, 0, Math.PI * 2); ctx.fill(); const lastPoint = simulation.points[simulation.points.length - 1]; ctx.fillStyle = 'red'; ctx.beginPath(); ctx.arc(lastPoint.x * 5 + 300, lastPoint.y * 5 + 200, 5, 0, Math.PI * 2); ctx.fill(); // إضافة معلومات ctx.fillStyle = 'black'; ctx.font = '14px Arial'; ctx.fillText(`نقاط المسار: ${simulation.points.length}`, 10, 20); ctx.fillText(`البداية (أخضر) - النهاية (أحمر)`, 10, 40); } else { // رسالة عندما لا توجد بيانات ctx.fillStyle = 'black'; ctx.font = '16px Arial'; ctx.textAlign = 'center'; ctx.fillText('لا توجد بيانات محاكاة. قم بإنشاء G-Code أولاً.', canvas.width / 2, canvas.height / 2); } }
  • ✅ الآن يعرض Top View المسار الفعلي المستخدم في المحاكاة
  • ✅ يدعم تحديث الرسم عند تغيير بيانات المحاكاة
  • ✅ يعرض نقاط البداية والنهاية بشكل واضح
  • ✅ يظهر معلومات عن عدد نقاط المسار

تحسين وظيفة generateRealGcode()

تم تحسين وظيفة generateRealGcode() لمعالجة بيانات الصورة بشكل صحيح وإنشاء G-Code أكثر دقة.

// التحسينات الرئيسية: function generateRealGcode(imageData, settings) { const { width, height, data } = imageData; const gcode = []; // إضافة رأس G-Code gcode.push('; G-Code generated by CNC AI'); gcode.push(`; Image dimensions: ${width}x${height}`); gcode.push('G21 ; Set units to millimeters'); gcode.push('G90 ; Use absolute positioning'); gcode.push('G17 ; Select XY plane'); gcode.push(`S${settings.spindleSpeed} ; Set spindle speed`); gcode.push('M3 ; Start spindle clockwise'); gcode.push('G4 P2 ; Wait for spindle to reach speed'); gcode.push(`F${settings.feedRate} ; Set feed rate`); gcode.push(''); // معالجة بيانات الصورة بكفاءة const threshold = settings.sensitivity * 25; const points = []; for (let y = 0; y < height; y += settings.stepOver) { let isCutting = false; let lastX = -1; for (let x = 0; x < width; x++) { const idx = (y * width + x) * 4; const avg = (data[idx] + data[idx + 1] + data[idx + 2]) / 3; if (avg < threshold) { if (!isCutting) { // بدء القطع points.push({ x, y, start: true }); isCutting = true; } lastX = x; } else { if (isCutting) { // إنهاء القطع points.push({ x: lastX, y, end: true }); isCutting = false; } } } if (isCutting) { points.push({ x: lastX, y, end: true }); } } // تحويل النقاط إلى أوامر G-Code let isToolDown = false; for (const point of points) { const x = (point.x / width) * settings.workAreaX; const y = (point.y / height) * settings.workAreaY; if (point.start) { if (isToolDown) { gcode.push('G0 Z2 ; Retract tool'); isToolDown = false; } gcode.push(`G0 X${x.toFixed(3)} Y${y.toFixed(3)} ; Rapid move to start`); gcode.push(`G1 Z-${settings.carvingDepth.toFixed(3)} F500 ; Plunge into material`); isToolDown = true; } else if (point.end) { if (isToolDown) { gcode.push('G0 Z2 ; Retract tool'); isToolDown = false; } gcode.push(`G0 X${x.toFixed(3)} Y${y.toFixed(3)} ; Move to end point`); } } // إضافة أوامر الختام gcode.push(''); gcode.push('G0 Z10 ; Retract tool fully'); gcode.push('M5 ; Stop spindle'); gcode.push('G0 X0 Y0 ; Return to home'); gcode.push('M30 ; End program'); return gcode.join('\n'); }
  • ✅ معالجة أكثر كفاءة لبيانات الصورة
  • ✅ تحسين خوارزمية كشف الحواف
  • ✅ إنشاء مسارات قطع أكثر سلاسة وكفاءة
  • ✅ تقليل وقت القطع عن طريق تحسين مسار الأداة

تحسين الأداء باستخدام Web Workers

تم تحسين أداء المعالجة باستخدام Web Workers لمنع تجميد واجهة المستخدم أثناء المعالجة الثقيلة.

// مثال على استخدام Web Worker: function processImageWithWorker(imageData, settings) { return new Promise((resolve) => { const worker = new Worker('image-processor.js'); worker.postMessage({ imageData, settings }); worker.onmessage = function(e) { if (e.data.type === 'progress') { updateProgress(e.data.value); } else if (e.data.type === 'result') { resolve(e.data.result); worker.terminate(); } }; }); }
  • ✅ منع تجميد واجهة المستخدم أثناء المعالجة
  • ✅ إمكانية معالجة صور أكبر دون مشاكل في الأداء
  • ✅ تقدم المعالجة يمكن عرضه للمستخدم
  • ✅ كود أكثر تنظيماً وسهولة في الصيانة
